def impute_age(cols):
    Age = cols[0]
    Pclass = cols[1]
    
    if pd.isnull(Age):
        
        if Pclass ==1:
            return avg_age_first_class
        elif Pclass ==2:
            return avg_age_second_class
        else:
            return avg_age_third_class
    else:
        return Age
